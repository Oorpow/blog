    # 1. 闭包

    # 1.1 什么是闭包

    • closure是一个函数以及其捆绑的词法环境的引用的组合。
    • 换而言之，closure可以从内部函数访问外部函数的作用域。
    • 函数执行导致函数被定义(函数表达式 | 函数声明)
    function fn(cb) {
    var n = 0
    cb()
}

function test() {
    console.log(n)  // n is not defined
}

fn()
// 此处并非closure，因此 test()并不能获取到 变量n
// 若是因为，此时可以称为closure

    # 1.2 常见形式

    1. 函数返回值是函数
    2. 函数返回的变量是函数
    3. 全局变量定义的闭包
    // 1. 函数返回值是函数
function fn() {
    let a = 3
    return function() {
        console.log(a)
    }
}

// 2. 函数返回的变量是函数
function fn() {
    let a = 3
    let n = function() {
        console.log(a)
    }
    return n
}
fn()()

// 3. 全局变量定义的闭包
let outter
function fn() {
    let a = 10
    outter = function() {
        log(a)
    }
}
fn()
outter()

// 4. 函数的参数为一个函数
let inner = function(fn) {
    fn()
}
function fn() {
    let a = 1
    let n = function() {
        log(a)
    }
    inner(n)
}

// 5. 迭代器
var add = (function() {
    var count = 0
    return function() {
        return ++count
    }
})()

add()   // 1
add()   // 2

    # 1.3 经典问题

    循环赋值

    function fn() {
    var arr = []
    for(var i = 0; i < 5; i++) {
        arr[i] = function() {
            log(i)
        }
    }
    return arr
}

let bar = fn()
for(var j = 0; j < 5; j++) {
    bar[j]()
}
// 5 5 5 5 5
// line80的 for循环中，var不存在块级作用域，循环体内给arr下标0-4的位置全赋值为一个函数
// line81 ~ line83 构成 closure，因此这5个closure 均可以访问到外部的 i，循环结束后，i = 5
// 此时line90执行后，均打印 5

    正确的循环赋值

    function fn() {
    var arr = []
    for(var i = 0; i < 5; i++) {
        arr[i] = (function(j) {
            return function() {
                log(j)
            }
        })(i)
    }
    return arr
}

let bar = fn()
for(var j = 0; j < 5; j++) {
    bar[j]()
}

// 改造为IIFE后，for循环每走一步，都将i传入，j得到的就是i的值，内部函数再读取外层作用域的j，从而打印出0 ~ 4

    # 1.4 试题

    function fn(n, o) {
    console.log(o)
    return {
        fn: function (m) {
            return fn(m, n)
        }
    }
}

var a = fn(0)   // n = 0, o = undefined, closure fn一直引用着n, n = 0
a.fn(1)         // fn(1, 0) n = 1, o = 0
a.fn(2)         // fn(2, 0) n = 2, o = 0
a.fn(3)         // fn(3, 0) n = 3, o = 0


var b = fn(0)   // n = 0, o = undefined
    .fn(1)      // fn(1, 0), n = 1, o = 0
    .fn(2)      // fn(2, 1), n = 2, o = 1
    .fn(3)      // fn(3, 2), n = 3, o = 2

var c = fn(0).fn(1)     // n = 0, o = undefined, fn(1) => fn(1, 0), n = 1, o = 0
c.fn(2)                 // fn(2) => fn(2, 1), n = 2, o = 1
c.fn(3)                 // fn(3) => fn(3, 1), n = 3, o = 1
